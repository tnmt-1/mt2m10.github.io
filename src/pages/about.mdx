export function BasicLabel({ label, imagePath }) {
  const boxStyle = {
    fontSize: 16,
    fontWeight: "bold",
    display: "inline-block",
    margin: "20px",
    height: "200px",
  };
  const imageAreaStyle = {
    marginBottom: "20px",
    paddingBottom: "10px",
    height: "110px",
    display: "flex",
    alignItems: "center",
  };
  const labelAreaStyle = {
    textAlign: "center",
  };
  const imageStyle = {
    width: "100px",
    maxHeight: "100%",
  };
  return (
    <div style={boxStyle}>
      <div style={imageAreaStyle}>
        <img src={imagePath} alt="no image" style={imageStyle} />
      </div>
      <div style={labelAreaStyle}>{label}</div>
    </div>
  );
}

export function LanguageLabel({ label, imagePath }) {
  return <BasicLabel label={label} imagePath={imagePath} />;
}

export function FrameworkLabel({ label, imagePath }) {
  return <BasicLabel label={label} imagePath={imagePath} />;
}

export function DatabaseLabel({ label, imagePath }) {
  return <BasicLabel label={label} imagePath={imagePath} />;
}

export function CloudLabel({ label, imagePath }) {
  return <BasicLabel label={label} imagePath={imagePath} />;
}

export function CareerPage({ title, start, end, roles, contents, skills }) {
  const sectionStyle = {
    border: "1px solid rgb(255, 255, 255, 0.2)",
    borderRadius: "10px",
    margin: "10px",
    padding: "30px",
  };
  const dateStyle = {
    marginBottom: "20px",
    display: "inline-block",
  };
  return (
    <section style={sectionStyle}>
      <h3 id={title}>{title}</h3>
      <small style={dateStyle}>
        {start} - {end}
      </small>
      <h4>役割</h4>
      <ul>
        {roles.map((role) => {
          return <li key={role}>{role}</li>;
        })}
      </ul>
      <h4>内容</h4>
      <ul>
        {contents.map((content) => {
          return <li key={content}>{content}</li>;
        })}
      </ul>
      <h4>技術・開発</h4>
      {skills.map((skill) => {
        return <CareerSkill key={skill} label={skill} />;
      })}
    </section>
  );
}

export function CareerSkill({ label }) {
  const style = {
    fontSize: "85%",
    border: "1px solid rgb(255, 255, 255, 0.2)",
    borderRadius: "20px",
    textAlign: "center",
    display: "inline-block",
    padding: "5px 10px",
    backgroundColor: "#d4b10a",
    color: "#000",
    fontWeight: "bold",
    marginRight: "15px",
  };
  return <div style={style}>{label}</div>;
}

export function CareerSubTitle({ title }) {
  const style = {
    borderBottom: "1px solid rgb(255, 255, 255, 0.4)",
    paddingBottom: "5px",
    marginTop: "60px",
  };
  return <h3 style={style}>{title}</h3>;
}
