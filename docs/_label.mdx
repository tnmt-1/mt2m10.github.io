export function BasicLabel({ label, imagePath }) {
  const boxStyle = {
    fontSize: 16,
    fontWeight: "bold",
    display: "inline-block",
    margin: "20px",
    height: "200px",
  };
  const imageAreaStyle = {
    marginBottom: "20px",
    paddingBottom: "10px",
    height: "110px",
    display: "flex",
    alignItems: "center",
  };
  const labelAreaStyle = {
    textAlign: "center",
  };
  const imageStyle = {
    width: "100px",
    maxHeight: "100%",
  };
  return (
    <div style={boxStyle}>
      <div style={imageAreaStyle}>
        <img src={imagePath} alt="no image" style={imageStyle} />
      </div>
      <div style={labelAreaStyle}>{label}</div>
    </div>
  );
}

export function LanguageLabel({ label, imagePath }) {
  return <BasicLabel label={label} imagePath={imagePath} />;
}

export function FrameworkLabel({ label, imagePath }) {
  return <BasicLabel label={label} imagePath={imagePath} />;
}

export function DatabaseLabel({ label, imagePath }) {
  return <BasicLabel label={label} imagePath={imagePath} />;
}

export function CloudLabel({ label, imagePath }) {
  return <BasicLabel label={label} imagePath={imagePath} />;
}
